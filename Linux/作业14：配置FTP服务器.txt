ftp配置：

#yum clean all

#yum install vsftpd -y

#yum install ftp -y

#vim /etc/vsftpd/vsftpd.conf


anonymous_enable=YES	//允许匿名访问

write_enable=YES	//全局性设置，开启写权限

anon_umask=022		//设置匿名用户新增文件的umask掩码权限

anon_upload_enable=YES		//允许匿名用户上传文件

anon_mkdir_write_enable=YES	//允许匿名用户新建目录

新增anon_other_write_enable=YES	//允许匿名用户其它权限（删除、重命名等）

新增anon_world_readable_only=NO	//允许匿名用户下载不具备读权限的文件

新增ftpd_banner=欢迎使用刘震老师的FTP服务器！	//设置欢迎词

:wq

#setup放行ftp

#setenforce 0	//放行安全策略

#service vsftpd restart
